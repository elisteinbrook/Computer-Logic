// This pseudocode segment is intended to compute and display
// the cost of home ownership for any number of users
// The program ends when a user enters 0 for the mortgage payment
start
   Declarations
      num mortgagePayment
      num utilities
      num taxes
      num upkeep
      num total
   Call startUp()
   while mortgagePayment <> 0
     Call MainLoop()
	 Call startUp() //added call startUp so program can continue loop if needed
   endwhile
   Call finishUp()
stop

 Procedure startUp()
   output "Enter your mortgage payment or 0 to quit"
   input mortgagePayment // changed "mtg" to the correct declaration
end procedure // changed return to end procedure

Procedure MainLoop()
   output "Enter utilities"
   input utilities
   output "Enter taxes"
   input taxes
   output "Enter amount for upkeep"
   input upkeep
   total = mortgagePayment + utilities + taxes + upkeep
   output "Total is ", total

end procedure // changed return to end procedure

Procedure finishUp()
   output "End of program"
end procedure // changed return to end procedure
